KISSY Pagination Demo

活生生的大栗子

默认参数调用

    new Pagination('#demo-1 .p-con').on('switch', function(e){
        S.one('#demo-1 .p-log').html('当前跳转到了第'+e.targetPage+'页');
    })

自定义参数调用

    var pagination= new Pagination('#demo-1 .p-con', {
        currentPage: 2,
        totalPage: 15
    }).on('switch', function(e) {
        S.one('#demo-1 .p-log').html('当前跳转到了第'+e.targetPage+'页');
    });

参数配置：

    {
        /**
         * 属性：当前页数
         * @type {number} 传入当前显示第几页，默认值为1。
         */
        currentPage:1,

        /**
         * 属性：总页数
         * @type {number} 传入页码总数，默认值为10。
         */
        totalPage:10,

        /**
         * 属性：样式表
         * @type {string} 如果想引入自定义样式，可以把样式表URL配置到这里。默认为组件自带样式。
         */
        styleSheet:'http://',

        /**
         * 属性：翻页器样式class前缀
         * @type {string} 可以自定义class前缀，默认为“y-”。
         */
        clsPrefix:'y-'
    }

方法调用：

    /**
     * 销毁pagination对象
     * @return {boolean} 如果成功销毁则返回true，否则返回false。
     */
    destroy();
    
    /**
     * 获取当前所在页数
     * @return {number} 返回当前所在的页码数。
     */
    getCurrentPage();
    
    /**
     * 显示页码
     * @return {object} 返回对象本身，可以链式调用。
     */
    show();
    
    /**
     * 隐藏页码
     * @return {object} 返回对象本身，可以链式调用。
     */
    hide();
    
    /**
     * 跳转到指定页
     * @param  {number} pageNum
     * @return {object} 返回对象本身，可以链式调用。
     */
    gotoPage(pageNum);
    
    /**
     * 跳转到下一页
     * @return {object} 返回对象本身，可以链式调用。
     */
    gotoNext();
    
    /**
     * 跳转到上一页
     * @return {object} 返回对象本身，可以链式调用。
     */
    gotoPrev();
    
    /**
     * 跳转到第一页
     * @return {object} 返回对象本身，可以链式调用。
     */
    gotoFirst();
    
    /**
     * 跳转到最后一页
     * @return {object} 返回对象本身，可以链式调用。
     */
    gotoLast();
